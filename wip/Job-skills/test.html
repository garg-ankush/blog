import pandas as pd
import ast
from helpers import get_top_counts
import plotly.express as px

def parse_list(s):
    try:
        return ast.literal_eval(s)
    except:  # noqa: E722
        return s

df = pd.read_csv("cleaned_up.csv", converters={
    'key_skills': parse_list,
    'tech_stack': parse_list,
    'languages': parse_list,
    'strong_experience': parse_list
})

test_df = pd.DataFrame([dict(get_top_counts(df['tech_stack']))]).T
